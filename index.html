function calcularIMC() { let nome = window.prompt('Digite o nome da pessoa:') let peso = Number(window.prompt(`Digite o peso de ${nome} em kg:`)) let altura = Number(window.prompt(`Digite a altura de ${nome} em metros:`)) let imc = peso / (altura * altura) let res1 = document.getElementById('resultado1') res1.innerHTML = `IMC de ${nome}:
` res1.innerHTML += `Peso: ${peso} kg, Altura: ${altura} m
` res1.innerHTML += `IMC: ${imc.toFixed(2)}
` if(imc < 18.5) { res1.innerHTML += `Classificação: Abaixo do peso` } else if(imc < 24.9) { res1.innerHTML += `Classificação: Peso normal` } else if(imc < 29.9) { res1.innerHTML += `Classificação: Sobrepeso` } else { res1.innerHTML += `Classificação: Obesidade` } }
